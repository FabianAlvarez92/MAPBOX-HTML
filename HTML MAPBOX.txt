    <script>
        // Token público de Mapbox
        mapboxgl.accessToken = 'pk.eyJ1IjoiZmFiaWFuYWx2YXJlejkyIiwiYSI6ImNtOTh5d2g4OTA4OXQycnE3Y2ZzamRiNGQifQ.89wshHZiPB7QhwllfgSkLw';

        // Crear mapa
        var map = new mapboxgl.Map({
            container: 'map',
            style: 'mapbox://styles/mapbox/streets-v11',
            center: [-74.5, 40],
            zoom: 9
        });

        // Marcador inicial con popup
        new mapboxgl.Marker()
            .setLngLat([-74.5, 40])
            .setPopup(new mapboxgl.Popup().setHTML("<h3>Ubicación inicial</h3><p>Este es un marcador.</p>"))
            .addTo(map);

        // Función para centrar en la ubicación del usuario
        function centrarUsuario() {
            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(function(position) {
                    const lat = position.coords.latitude;
                    const lng = position.coords.longitude;
                    map.setCenter([lng, lat]);
                    new mapboxgl.Marker()
                        .setLngLat([lng, lat])
                        .setPopup(new mapboxgl.Popup().setHTML("<h3>Estás aquí</h3>"))
                        .addTo(map);
                }, function(error) {
                    alert("No se pudo obtener tu ubicación.");
                });
            } else {
                alert("La geolocalización no está disponible en este dispositivo.");
            }
        }
    </script>
